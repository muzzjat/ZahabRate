<script>
    // قاعدة بيانات الأسعار حسب الدولة
    const marketData = {
        "UAE": { title: "سوق الإمارات", rates: [["USD", "3.66", "3.68"], ["INR", "0.043", "0.045"]] },
        "SA": { title: "سوق السعودية", rates: [["USD", "3.75", "3.76"], ["EGP", "0.078", "0.082"]] },
        "EG": { title: "سوق مصر", rates: [["USD", "47.50", "48.00"], ["EUR", "51.00", "52.00"]] }
    };

    function updateData() {
        const country = document.getElementById('country').value;
        const data = marketData[country];
        document.getElementById('marketTitle').innerText = data.title;
        const body = document.getElementById('rateBody');
        body.innerHTML = data.rates.map(r => `<tr><td>${r[0]}</td><td>${r[1]}</td><td>${r[2]}</td></tr>`).join('');
    }

    // تشغيل عند التحميل
    updateData();
</script>
